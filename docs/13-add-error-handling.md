# Add error handling

1. Add `error` state in `src/lib/store.js`

   ```diff
     const state = {
       currentUser: null,
   +   error: null,
     };

     const getters = {
       isLoggedIn(state) {
         return state.currentUser !== null;
       },

   +   hasError(state) {
   +     return state.error !== null;
   +   },
     };

     const mutations = {
       setCurrentUser(state, user) {
         state.currentUser = user;
       },

   +   setError(state, message) {
   +     state.error = message;
   +   },
   +
   +   clearError(state) {
   +     state.error = null;
   +   },
     };
   ```
